                                        peek();

                                        isFull();

                                    case 5:
                                        empty();
                                        break;

                                        return 0;
                                        
                                    default:
                                        printf("~ Enter a Valid Input .");
                                }
                            }
                                return 0;
                            }
                            
                            void push()
                            {
                                if(top == 100)
                                {
                                    printf("~ Overflow..");
                                    return ;
                                }
                                
                                int item;
                                printf("~ Enter the item to push : ");
                                scanf("%d", &item);
                                
                                top = top + 1;
                                stack[top] = item; 
                                
                                printf("~ Added The Element!");
                                return;
                            }
                            
                            void pop()
                            {
                                if(top == -1)
                                {
                                    printf("~ Underflow.");
                                    return ;
                                }
                                
                                top = top - 1;
                                printf("~ Element Popped.");
                                return ;
                            }
                            
                            void peek()
                            {
                                if(top == -1)
                                {
                                    printf("~ No Element is in The STack yet !");
                                    return;
                                }
                                printf("~ Peek Element is %d", stack[top]);
                                return ;
                            }
                            
                            void isFull()
                            {
                                if(top == 100)
                                {
                                    printf("~ It is Full.");
                                    return;
                                }
                                else{
                                    printf("~ It is not Full.");
                                }
                            }
                            
                            void empty()
                            {
                                if(top == -1)
                                {
                                    printf("~ It is Empty.");
                                    return;
                                }
                                else{
                                    printf("~ It is not Empty.");
                                    return ;
                                }
                                
                            }

                            struct Node{
                                int val;
                                struct Node *next;
                            };
                            
                            
                            struct Node *head;
                            
                            int main()
                            {
                                printf("\n **********************Menu Options*************************\n\n");
                                printf("Push \t\t\t--\t\t (press 1)");
                                printf("\nPop  \t\t\t--\t\t (Press 2)");
                                printf("\nPeek  \t\t\t--\t\t (Press 3)");
                                printf("\nisFull  \t\t--\t\t (Press 4)");
                                printf("\nEmpty  \t\t\t--\t\t (Press 5)");
                                printf("\nExit  \t\t\t--\t\t (Press 0)");
                                printf("\n _____________________________________________________________\n\n");

                            {
                                    
                                int choice;

                                {
                                    case 1: 
                                        push();

                                    case 2: 
                                        pop();
                                        break;
                                    
                                    case 3:
                                        peek();
                                        break;
                                    case 4:
                                        isFull();
                                        break;
                                        
                                    case 5:
                                        Empty();
                                        break;
                                                
                                    case 0:
                                        return 0;
                                        
                                    default:
                                        printf("~ Enter a Valid Input .");
                                }
                            }
                                return 0;
                            }
                            
                            void push()
                            {
                                int item;
                                printf("~ Enter The item : ");
                                scanf("%d", &item);
                                
                                struct Node *newNode;
                                newNode = (struct Node *)malloc(sizeof(struct Node));
                                
                                newNode->val = item;
                                newNode->next = NULL;
                                if(head == NULL)
                                {	
                                    head = newNode;
                                }
                                else
                                {
                                    newNode->next = head;
                                    head = newNode;
                                }
                                printf("~ Element is Added !");
                                return;
                            }
                            
                            void pop()
                            {
                                if(head == NULL)
                                {
                                    printf("~ UnderFlow!");
                                    return;
                                }
                                
                                head = head->next;
                                printf("~ Element is Popped !");
                                return ;
                            }
                            
                            void peek()
                            {
                                if(head == NULL)
                                {
                                    printf("~ No Element is in The Stack Yet !");
                                    return ;
                                }
                                printf("~ Peek Element is : %d",head->val);
                                return;
                            }
                            
                            void Empty()
                            {
                                if(head == NULL)
                                {
                                    printf("~ It is Empty!");
                                }
                                else{
                                    printf("~ It is not Empty!");
                                }
                            }

                                printf("~ It is not Full.");
